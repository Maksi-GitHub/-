#include <iostream>
#include <vector>

using namespace std;

 //Задание 1
 double CalculateS(int n)
 {
     double sum = 0.0;
 
     for (int i = 0; i <= n; i++)
     {
         double term = 1.0 / pow(2 * i + 1, 2);
         sum += term;
     }
 
     return sum;
 }
 
 int main()
 {
     setlocale(LC_ALL, "rus");
     cout << "Введите натуральное число n: ";
     int n;
     cin >> n;
 
     double result = CalculateS(n);
 
     cout << "Значение S для n = " << n << ": " << result << endl;
 
     return 0;
 }
